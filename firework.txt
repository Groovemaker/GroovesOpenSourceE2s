@name Runics Firework
@inputs 
@outputs 
@persist Teig:entity Apply
@trigger 

if(first() || duped()){
    Apply = 0
    timer("usesystem",10)
    Teig = propSpawn("models/props_c17/canister02a.mdl",ang(0,0,0),1)
}

if(clk("usesystem")){
    Allps = players()
    for(I=1,Allps:count()){
        Guy = Allps[I,entity]
        if(Guy:keyUse() && Guy:aimEntity() == Teig){
            Teig:propFreeze(0)

            Apply = 1
            Teig:setTrails(30,30,120,"trails/smoke",vec(255,255,255), 255)
            timer("fly",10)
            timer("splode",2000)
        }
    }

    timer("usesystem",440)
}
if(clk("fly")){
    if(Apply == 1){
        Teig:applyForce(vec(0,0,100000))

    }
    timer("fly",2000)
}
if(clk("splode")){
    Killa = propSpawn("models/props_phx/cannonball.mdl",ang(0,0,0),1)
    Killa:setPos(Teig:pos())
    Killa:propBreak()
    Apply = 0
    for(I=1,55){
        holoCreate(I)
        holoPos(I,Teig:pos())
        holoAlpha(I,0)
        holoAng(I,randvec(-360,360):toAngle())
        holoEntity(I):setTrails(220,220,120,"trails/physbeam",vec(randint(11,255),randint(11,255),randint(11,255)), 255)
    }
    Teig:propDelete()
    #Teig:setAlpha(0)
    timer("cleanup",7000)
}

if(clk("cleanup")){
    selfDestructAll()
}
for(I=1,55){
    holoPos(I,holoEntity(I):toWorld(vec(0,100,0)))
}
